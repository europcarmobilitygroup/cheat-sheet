            \sheetTag{fix}{90} \sheetTagText{patches a bug in the code of the non-production environment \faCodeBranch $\rightarrow$ \inlinecode{Java}{develop}} \par
            \sheetTag{hotfix}{85} \sheetTagText{backward-compatible bug fix for the production environment \faCodeBranch $\rightarrow$ \inlinecode{Java}{master}} \par
            \sheetTag{perf}{80} \sheetTagText{any performance improvements, including the code splitting} \par
            \sheetTag{test}{75} \sheetTagText{adding new or refactoring existing \inlinecode{Java}{.spec} files: unit, snapshot or integration tests} \par
            \sheetTag{refactor}{70} \sheetTagText{changes such as: simplifying or removing redundant code, renaming variables} \par
            \sheetTag{style}{65} \sheetTagText{any styling changes to \inlinecode{Java}{.css} or \inlinecode{Java}{.scss} file or \inlinecode{Java}{<style lang="scss">}} \par
            \sheetTag{docs}{60} \sheetTagText{new documentation for the developers, both: inline \inlinecode{Java}{JSDoc} and markdown files \inlinecode{Java}{.md}} \par
            \sheetTag{ci}{55} \sheetTagText{changes related to the continuous integration and deployment: scripts, config and tools} \par
